const customStyles = {
  overlay: {

  },
  content : {
    top                   : '50%',
    left                  : '50%',
    right                 : 'auto',
    bottom                : 'auto',
    marginRight           : '-50%',
    transform             : 'translate(-50%, -50%)'
  }
};
 
 
class ExampleModal extends React.Component {
  constructor(props) {
    super(props);
    // console.log(props);
    this.state = {
      modalIsOpen: this.props.modalIsOpen
    };
  }
 
  render() {
    return (
      <div>
        <Modal

          loop={true}
        />
 
      </div>
    );
  }
}
 
export default ExampleModal;
